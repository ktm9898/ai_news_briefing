🔗 Apps Script URL

구글 시트 > 확장 프로그램 > Apps Script에서 배포한 URL

⚙️ 초기 설정 안내

이 대시보드가 구글 시트의 뉴스 데이터를 읽으려면 Google Apps Script를 한 번만 설정하면 됩니다.

  1. 구글 시트를 열고 확장 프로그램 > Apps Script 클릭
  2. 아래 코드를 붙여넣고 저장
  3. 배포 > 새 배포 → 유형: 웹 앱 → 액세스: 모든 사용자
  4. 생성된 URL을 복사하여 아래 입력란에 붙여넣기
function doGet(e) {
  var tab = (e && e.parameter.tab) || "News_Data";
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var ws = ss.getSheetByName(tab);
  var data = ws.getDataRange().getValues();
  var headers = data[0];
  var result = [];
  for (var i = 1; i < data.length; i++) {
    var row = {};
    for (var j = 0; j < headers.length; j++) {
      row[headers[j]] = data[i][j];
    }
    result.push(row);
  }
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var body = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var ws = ss.getSheetByName("Settings");
  if (body.action === "add") {
    ws.appendRow([body.category, body.keyword, "TRUE"]);
  } else if (body.action === "delete") {
    ws.deleteRow(body.row + 1); // +1 for header
  } else if (body.action === "toggle") {
    var current = ws.getRange(body.row+1, 3).getValue();
    ws.getRange(body.row+1, 3).setValue(
      current === "TRUE" ? "FALSE" : "TRUE"
    );
  }
  return ContentService
    .createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
